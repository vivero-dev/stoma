    @vivero/stoma

    Stoma - declarative API gateway as a TypeScript library.

    Built on Hono for Cloudflare Workers and edge runtimes. Define routes, compose policies (auth, rate limiting, caching, transforms), and proxy to upstream services - all from a single configuration object.

    Example

    import {
      createGateway,
      cors,
      jwtAuth,
      rateLimit,
      cache,
      requestLog,
      health,
    } from "@vivero/stoma";

    const gateway = createGateway({
      name: "my-api",
      basePath: "/api",
      debug: env.DEBUG,
      policies: [cors(), requestLog(), rateLimit({ max: 100 })],
      routes: [
        health(),
        {
          path: "/users/*",
          pipeline: {
            policies: [
              jwtAuth({ secret: env.JWT_SECRET }),
              cache({ ttlSeconds: 60 }),
            ],
            upstream: { type: "url", target: "https://users.internal" },
          },
        },
      ],
    });

    export default gateway.app;
